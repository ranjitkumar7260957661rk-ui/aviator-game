const express = require("express"); const app = express(); app.use(express.json()); let users = { "user1": { balance: 1000 } }; let requests = []; // 🎮 Spin API app.post("/spin", (req, res) => { let { user, bet } = req.body; if (!users[user] || users[user].balance < bet) { return res.json({ error: "Low balance" }); } users[user].balance -= bet; let rand = Math.random() * 100; let multiplier = 0; if (rand < 35) multiplier = 0; else if (rand < 60) multiplier = 0.5; else if (rand < 80) multiplier = 1; else if (rand < 92) multiplier = 2; else if (rand < 98) multiplier = 5; else multiplier = 10; let win = bet * multiplier; users[user].balance += win; res.json({ multiplier, win, balance: users[user].balance }); }); // 💸 Deposit Request app.post("/deposit", (req, res) => { let { user, amount } = req.body; requests.push({ type: "deposit", user, amount, status: "pending" }); res.json({ msg: "Deposit request submitted" }); }); // 💰 Withdraw Request app.post("/withdraw", (req, res) => { let { user, amount } = req.body; if (users[user].balance < amount) { return res.json({ error: "Low balance" }); } users[user].balance -= amount; requests.push({ type: "withdraw", user, amount, status: "pending" }); res.json({ msg: "Withdraw request submitted" }); }); // 👤 Balance check app.get("/balance/:user", (req, res) => { res.json(users[req.params.user]); }); app.listen(3000, () => console.log("Server running on 3000"));
